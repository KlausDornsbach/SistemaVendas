CREATE TABLE VENDA
(CLIENTE_CPF INTEGER,
 PRODUTO_UPC INTEGER,
 DATA INTEGER,
 HORA INTEGER,
 DESCONTO INTEGER,
 TIPO_PAGAMENTO VARCHAR(10),
 ID INTEGER,
 PRIMARY KEY(ID),
 FOREIGN KEY(CLIENTE_CPF) REFERENCES CLIENTE(CPF),
 FOREIGN KEY(PRODUTO_UPC) REFERENCES PRODUTO(UPC)
);

CREATE TABLE CLIENTE
(CPF INTEGER,
 NOME VARCHAR(25),
 ENDERECO VARCHAR(25)
 PRIMARY KEY(CPF)
);

CREATE TABLE PRODUTO
(UPC INTEGER,
 VALOR INTEGER,
 DESCRICAO VARCHAR(25),
 PRIMARY KEY(UPC)
);
